### 1. 角色说明

| 角色   | 唯一性  | 说明               |
| ------ | ------- | ------------------ |
| owner  | 每组1人 | 创建者，最高权限   |

### 1. 创建组（createGroup/index.js）

javascript

const agconnect = require('@agconnect/database');

const db = agconnect.database();

const { generateInviteCode, uuid } = require('../utils');

exports.handler = async (event) => {

const { groupName, userId } = event.body;

const now = new Date();

const groupId = uuid();

try {

// 1. 创建组

await db.collection('Group').add({

id: groupId,

name: groupName,

creatorId: userId,

inviteCode: generateInviteCode(),

inviteCodeExpireTime: new Date(now.getTime() + 24 _ 60 _ 60 \* 1000), // 24小时有效期

enabledFeatures: ['schedule'],

featurePermissions: JSON.stringify({

schedule: { member: 'self', admin: 'all', owner: 'all' }

}),

createTime: now

});

// 2. 创建者加入组(owner角色)
await db.collection('GroupMember').add({
id: uuid(),
groupId: groupId,
userId: userId,
role: 'owner',
joinTime: now
});

return { code: 200, message: '创建成功', data: { groupId } };
} catch (err) {

console.error('创建组失败:', err);

return { code: 500, message: '创建失败' };

}

};

### 2. 加入组（joinGroup/index.js）

javascript

const agconnect = require('@agconnect/database');

const db = agconnect.database();

const { uuid } = require('../utils');

exports.handler = async (event) => {

const { inviteCode, userId } = event.body;

const now = new Date();

try {

// 1. 校验邀请码

const groups = await db.collection('Group')

.where('inviteCode').eq(inviteCode)

.get();

if (!groups.length) {
return { code: 404, message: '邀请码无效' };
}

const group = groups[0];
if (group.inviteCodeExpireTime < now) {
return { code: 410, message: '邀请码已过期' };
}

// 2. 校验是否已加入
const members = await db.collection('GroupMember')
.where({ groupId: group.id, userId })
.get();

if (members.length) {
return { code: 200, message: '已在群组中' };
}

// 3. 加入组
await db.collection('GroupMember').add({
id: uuid(),
groupId: group.id,
userId: userId,
role: 'member',
joinTime: now
});

return { code: 200, message: '加入成功' };
} catch (err) {

console.error('加入组失败:', err);

return { code: 500, message: '加入失败' };

}

};

### 3. 任命/取消管理员（setAdmin/index.js）

javascript

const agconnect = require('@agconnect/database');

const db = agconnect.database();

exports.handler = async (event) => {

const { groupId, operatorId, targetUserId, isAdmin } = event.body;

try {

// 1. 校验操作者是否为owner

const operators = await db.collection('GroupMember')

.where({ groupId, userId: operatorId })

.get();

if (!operators.length || operators[0].role !== 'owner') {
return { code: 403, message: '仅创建者可任命管理员' };
}

// 2. 校验目标用户
const targets = await db.collection('GroupMember')
.where({ groupId, userId: targetUserId })
.get();

if (!targets.length) {
return { code: 404, message: '用户不在群组中' };
}

if (targets[0].role === 'owner') {
return { code: 403, message: '不能修改创建者权限' };
}

// 3. 更新角色
await db.collection('GroupMember')
.doc(targets[0].id)
.update({ role: isAdmin ? 'admin' : 'member' });

return { code: 200, message: `已${isAdmin ? '任命' : '取消'}管理员` };
} catch (err) {

console.error('任命管理员失败:', err);

return { code: 500, message: '操作失败' };

}

};

### 4. 解散组（dismissGroup/index.js）

> 使用事务删除组所有关联数据，避免脏数据残留
> javascript

const agconnect = require('@agconnect/database');

const db = agconnect.database();

exports.handler = async (event) => {

const { groupId, operatorId } = event.body;

try {

// 1. 校验操作者是否为owner

const operators = await db.collection('GroupMember')

.where({ groupId, userId: operatorId })

.get();

if (!operators.length || operators[0].role !== 'owner') {
return { code: 403, message: '仅创建者可解散组' };
}

// 2. 开启事务删除所有关联数据
const transaction = db.startTransaction();

// 删除组成员
await transaction.collection('GroupMember')
.where({ groupId })
.delete();

// 删除排班数据
await transaction.collection('Schedule')
.where({ groupId })
.delete();

// 删除组本身
await transaction.collection('Group')
.doc(groupId)
.delete();

await transaction.commit();

return { code: 200, message: '解散成功' };
} catch (err) {

console.error('解散组失败:', err);

return { code: 500, message: '解散失败' };

}

};

## ❓ 常见问题排查

| 问题           | 解决方案                                                        |
| -------------- | --------------------------------------------------------------- |
| 部署表失败     | 检查JSON语法，确保`primaryKeys`字段正确，无重复字段             |
| 云函数调用失败 | 检查函数名是否正确，参数格式是否符合要求，AGC中云函数是否已启用 |
| 权限校验不生效 | 确保权限判断仅在云函数实现，端侧仅做UI显隐                      |
| 数据不同步     | 检查`belongPrimaryKey`配置，确保值为`groupId`                   |
| 邀请码无效     | 检查邀请码是否过期，是否被重置，云函数逻辑是否正确              |
| 真机装不上     | 检查UDID是否已注册到AGC，调试Profile是否包含该设备              |
